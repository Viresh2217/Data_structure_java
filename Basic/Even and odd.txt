package Practies.java;

import java.util.*;
class Tp{
public static void main(String args[])
{

	Scanner sc=new Scanner(System.in);
	
	System.out.println("Enter a number");
	
	int i=sc.nextInt();
	
	if(i%2==0) {
		System.out.println(i +" is Even Number");
	}
	else {
		System.out.println(i+" is Odd Number");
	}
	
	
}
}
